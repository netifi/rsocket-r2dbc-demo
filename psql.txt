CREATE TABLE records (
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    aliases varchar[],
    authors varchar[],
    description varchar,
    background varchar,
    thumbnail varchar,
    name varchar,
    partners varchar[],
    powers varchar[],
    comic_count int,
    event_count int,
    pageview_count int,
    serie_count int,
    story_count int,
    secret_identities varchar[],
    species varchar[],
    supername varchar,
    teams varchar[],
    marvel_url varchar,
    wikipedia_url varchar
);
