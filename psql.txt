CREATE TABLE records (
    id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    aliases varchar[],
    authors varchar[],
    description varchar,
    background varchar,
    thumbnail varchar,
    name varchar,
    partners varchar[],
    powers varchar[],
    comicCount int,
    eventCount int,
    pageviewCount int,
    serieCount int,
    storyCount int,
    secretIdentities varchar[],
    species varchar[],
    superName varchar,
    teams varchar[],
    marvelUrl varchar,
    wikipediaUrl varchar
);
